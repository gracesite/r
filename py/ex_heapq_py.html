'''
heapq - Pythonâ€™s heapq module implements a min-heap.
* Binary heaps implemented with zero-based array, parent at N, children are at index of 2*N + 1 and 2*N + 2 
* heapq.heapify(),heapq.heappush(), heapq.heappop()
* heapq.nsmallest(3, data), heapq.nlargest(3, data)
*** heapq.merge(*data) # it uses memory of num of array not the num of items
'''
import heapq
import random

data = []
result = []
for i in range(4):
    new_data = list(random.sample(range(1,101),5))
    new_data.sort()
    data.append(new_data)
print(data)
for i in heapq.merge(*data):
    result.append(i)
print(result)
