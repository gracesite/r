import argparse

parser = argparse.ArgumentParser()
parser.add_argument('-f', '--foo', default='foobar')
parser.add_argument('-b', '--bar', default=3.14)
args = parser.parse_args()

print('args.foo=', args.foo) 
print('args.bar=', args.bar)

# To run the program:
# python3 ex_arg_parse.py --foo gracefoot --bar 4.5
# Output:
# args.foo= gracefoot
# args.bar= 4.5
