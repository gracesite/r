'''
> python3 ex_argparse.py --input_file myfile.txt --start_date 12/01/2019 --end_date 02/11/2020
output:
args.inputfile= myfile.txt
args.start_date= 12/01/2019
args.end_date= 02/11/2020
'''
import argparse

parser = argparse.ArgumentParser()

parser.add_argument('--input_file', help='input file name')
parser.add_argument('--start_date', help='eg 12/01/2019')
parser.add_argument('--end_date', help='eg 02/11/2020')
args = parser.parse_args()
print('args=', args)
print('args.inputfile=', args.input_file)
print('args.start_date=', args.start_date)
print('args.end_date=', args.end_date)
