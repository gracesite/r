# Example Python program to plot a complex bar chart 
import math
import pandas as pd
import matplotlib.pyplot as plot
from itertools import cycle, islice

TITLE = "Bears Population Growth over the years"
YEARS = ["2018", "2019", "2020"]
XLABEL = 'Year'
YLABEL = 'Bears Population'

data = {"Black Bear":[7405, 7485, 7815],
        "Polar Bear":[3045, 3572, 3638],
        "Brown Bear":[4558,4677, 4810],
        "Gummy Bear":[5000,5100,5300]
        }

# Dictionary loaded into a DataFrame   
my_colors = list(islice(cycle(['green', 'red', 'yellow', 'blue']), None, len(data)))      
dataFrame = pd.DataFrame(data=data, index=YEARS)
fig = plot.figure()
ax = fig.add_subplot(111) # 1 plot
ax.set_xlabel(XLABEL)
ax.set_ylabel(YLABEL)

# calculate y-limit for legend
max_value = max([col for row in data.values() for col in row])
ylimit = (1000 * math.ceil(max_value / 1000)) + 1000
ax.set_ylim(0,ylimit) # set y limit


dataFrame.plot.bar(rot=15,  color=my_colors, ax=ax, title=TITLE)
# horizontal grid on y-axis
plot.grid(b=True, which='both', axis='y', color='#D3D3D3', linestyle='-')

num_columns = len(data.keys())
ax.legend(loc='upper center', bbox_to_anchor=(0.5, 1.0), prop={'size': 8}, shadow=False, ncol=num_columns)

plot.show()

# save the plot image into a file
fig.savefig('plot_image.png')
