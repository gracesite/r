''' 
deque is a Double-ended queue 
add from both ends: exend(), append(), exendleft(), appendleft()
pop from both end: pop(), popleft()
rotate: dq.rotate(2)  dq.rotate(-2)
deque(maxlen=5) #limit the size to 5 with maxlen
'''
from collections import deque

str1 = 'abcdefg'
dq = deque(str1)

# add to the right end
dq.extend('RR')
dq.append('K')
print(dq)

print('add to the left end')
# add to the left end
dq.extendleft('123')
dq.appendleft('0')
print(dq)

dq.pop()
dq.popleft()
print('dq after pop, popleft')
print(dq)
