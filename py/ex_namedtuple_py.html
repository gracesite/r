'''
standard tuple uses index to access members. eg: 
worker = ('Jane', 28, 'engineer')
worker[0], worker[1], worker[2]

namedtuple
_fields
_replace() - it is immutable, return a new object
'''
import collections

Person = collections.namedtuple('Person', 'name, age, title')
bob = Person('Bob', 30, 'engineer')
jan = Person('Jena', 28, 'vp')
print('object:',id(jan))
print('jan._fields=',jan._fields) #('name', 'age', 'title')
# _replace return a new namedtuple
jan = jan._replace(name='Janifer')
print('jan=', jan)
print('anoter object:', id(jan))

for p in [bob, jan]:
  print('{} is a {} year old {}'.format(*p))
